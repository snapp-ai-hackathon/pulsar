1. Get the application URL by running these commands:
  {{- if .Values.ingress.enabled }}
  echo "Visit http{{ if (index .Values.ingress.tls 0) }}s{{ end }}://{{ (index .Values.ingress.hosts 0).host }} to access Pulsar"
  {{- else if eq .Values.service.type "NodePort" }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath='{.spec.ports[0].nodePort}' services {{ include "pulsar.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath='{.items[0].status.addresses[0].address}')
  echo "Visit http://$NODE_IP:$NODE_PORT"
  {{- else if eq .Values.service.type "LoadBalancer" }}
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "pulsar.fullname" . }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
  echo "Visit http://$SERVICE_IP"
  {{- else }}
  echo "Use port-forwarding to access the service: kubectl port-forward svc/{{ include "pulsar.fullname" . }} 8088:{{ .Values.service.targetPort }}"
  {{- end }}
